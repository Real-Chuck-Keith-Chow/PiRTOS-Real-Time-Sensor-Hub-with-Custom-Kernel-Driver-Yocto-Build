cmake_minimum_required(VERSION 3.13)
include($ENV{PICO_SDK_PATH}/external/pico_sdk_import.cmake)

project(sensor_mvp C CXX ASM)
pico_sdk_init()

add_executable(sensor_mvp
    main.c
)

pico_enable_stdio_usb(sensor_mvp 1)
pico_enable_stdio_uart(sensor_mvp 0)

target_link_libraries(sensor_mvp
    pico_stdlib
    hardware_adc
    hardware_i2c
)

pico_add_extra_outputs(sensor_mvp)
